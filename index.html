Named regex captures

const regexp = /\/
  (?<hierarchy>[^\/]+)
  (?<membership>\/.*?)[\/]*
  (?<name>[^\/]*)
  $/u;

const hierarchicalName = "/processArea/MK/subsystem1/MainPage";
const { groups } = regexp.exec(hierarchicalName)
// {
//   hierarchy:  "processArea",
//   membership: "/MK/subsystem1",
//   name:       "MainPage",
// }
